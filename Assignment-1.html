# Loop through columns
for (colname in names(Typical_Employee_Survey_Data_3_)) {
  
  column_data <- Typical_Employee_Survey_Data_3_[[colname]]
  
  # If numeric: summary + boxplot
  if (is.numeric(column_data)) {
    cat("===== 5-Point Summary for", colname, "=====\n")
    print(fivenum(column_data, na.rm = TRUE))
    cat("\n")
    
    boxplot(column_data,
            main = paste("Boxplot of", colname),
            col = "lightblue")
    
    # If factor: counts + barplot
  } else if (is.factor(column_data)) {
    cat("===== Frequency Counts for", colname, "=====\n")
    print(table(column_data))
    cat("\n")
    
    barplot(table(column_data),
            main = paste("Barplot of", colname),
            col = "lightgreen")
  }
}
